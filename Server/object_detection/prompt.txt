To prevent redundant outputs when the scene hasn’t changed, we will maintain state between requests. Specifically, we store:

The last input image signature (e.g. a hash)
The last set of detected objects (classes and approximate positions or state)

    pass  # if no GPU, continue on CPU

# State variables
last_hash = None
last_detections = []

# Helper to parse results into simple dicts
def parse_detections(results):
    detections = []
    res = results[0]  # results is a list, take first frame
    boxes = res.boxes
    for i in range(len(boxes)):
        cls_id = int(boxes.cls[i])
        label = model.model.names[cls_id]  # class names from model
        conf = float(boxes.conf[i])
        xyxy = boxes.xyxy[i].tolist()     # [x1,y1,x2,y2]
        detections.append({"label": label, "conf": conf, "bbox": xyxy})
    return detections

# Main detection function

        return {"changed": False, "objects": []}
